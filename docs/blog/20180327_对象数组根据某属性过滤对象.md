**对象数组根据某属性进行排序显示**

```js
var infoObj = [
  {
    name: "张三",
    sex: 'female',
    age: 30,
  },
  {
    name: "李四",
    sex: 'male',
    age: 20,
  },
  {
    name: "王五",
    sex: 'female',
    age: 40,
  }];
 
function compare(propertyName) {
  return function (object1, object2) {
    var value1 = object1[propertyName];
    var value2 = object2[propertyName];
    return value1 - value2
  }
}
infoObj.sort(compare("age"))
```
